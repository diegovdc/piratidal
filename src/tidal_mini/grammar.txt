pattern = cat (<whitespace> <'.'> <whitespace> cat)*
cat = token (<whitespace>? token)*
<token> = word | number | silence | fast | group | alt | slow | iso
group = (<#'\['> stack <#'\]'> | token <'!'> number)
iso = <#'\{'> stack <#'\}'> iso-size?
iso-size = <'%'> number
fast?
alt = <#'\<'> stack <#'\>'>
fast?
fast = token <'*'> number
slow = token <'/'> number
whitespace = #"\s+"
word = #"[a-zA-Z]+"
stack = pattern (<whitespace> <','> <whitespace> pattern)*
number = #"[0-9]+"
silence = "~"
